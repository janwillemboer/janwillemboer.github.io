Spiekbrief Methoden 3

// eigen methode roep je aan zoals je gewend bent
// naam van de methode
//          parameters/argumenten tussen haakjes
//                                          puntkomma
toonRelatie("Hans", "Grietje", "broer en zus");

// eigen methode declaratie:
// public
//    returnwaarde (of void)
//           methodenaam
//                      tussen haakjes de variabelen waarin
//                      de parameters/argumenten worden opgeslagen
//                      voor deze methode
//                                             accolades voor method body
public void toonRelatie(String naam1, String naam2, String relatie) {
    SaxionApp.print(naam1);
    SaxionApp.print(" en ");
    SaxionApp.print(naam2);
    SaxionApp.print(" zijn ");
    SaxionApp.print(relatie);
    SaxionApp.printLine(" van elkaar.");
}

// een eigen methode kan ook een returnwaarde hebben
int inhoudKubus = berekenKubusInhoud(50);

// declaratie eigen methode met returnwaarde:
// public
//   returntype
//         naam methode
//                         parameters/argumenten
//                                  methode body tussen accolades
// in de methodbody:
//  return een waarde waarvan het type hetzelfde is als het returntype 
//  van de methode.
public int berekenKubusInhoud(int zijde) {
    int inhoud = zijde * zijde * zijde;
    return inhoud;
}

// accolades geven variabelen "scope". Buiten de accolades is de 
// variabele niet zichtbaar of bruikbaar.
public void run() {
    int getal = 10;
    SaxionApp.printLine(getal); // getal is binnen de scope, dus kan getoond worden
    andereMethode();
}

public void andereMethode() {
    // FOUT: dit kan niet, getal is niet binnen de scope van deze methode.
    getal = 11; 
}
